<script>
	$(document).ready(function() {
		$('#demo-thumb-yes:not(.selected)').click(function() {
			$('#demo-thumb-yes').addClass('selected');
			$('#demo-thumb-no').removeClass('selected');
		})
		$('#demo-thumb-no:not(.selected)').click(function() {
			$('#demo-thumb-yes').removeClass('selected');
			$('#demo-thumb-no').addClass('selected');
		})
	});
</script>
